\begin{tikzpicture}[scale=0.5]
    % Define coordinates
    \coordinate (p1) at (0,0);
    \coordinate (p2) at (4,4);
    \coordinate (p3) at (0,8);
    \coordinate (p4) at (4,12);
    \coordinate (p5) at (0,14);
    \coordinate (q) at (0,10);

    % Draw axes
    \draw[->] (-2,0) -- (6,0) node[right] {$x$};
    \draw[->] (0,-2) -- (0,16) node[above] {$y$};

    % Draw points
    \filldraw (p1) circle (2pt) node[below left] {$p_1$};
    \filldraw (p2) circle (2pt) node[below right] {$p_2$};
    \filldraw (p3) circle (2pt) node[above left] {$p_3$};
    \filldraw (p4) circle (2pt) node[above right] {$p_4$};
    \filldraw (p5) circle (2pt) node[above left] {$p_5$};
    \filldraw (q) circle (2pt) node[above left] {$q$};

    % Draw directed edges
    \draw[->, thick] (p1) -- (p2);
    \draw[->, thick] (p2) -- (p3);
    \draw[->, thick] (p3) -- (p4);
    \draw[->, thick] (p4) -- (p5);

    % Draw lune for (p3, p4)
    \path let \p1=(p3), \p2=(p4) in
        node[draw, circle, red, dashed, minimum size={2*veclen(\x2-\x1,\y2-\y1)}] at (\p1) {};
    \path let \p1=(p3), \p2=(p4) in
        node[draw, circle, red, dashed, minimum size={2*veclen(\x2-\x1,\y2-\y1)}] at (\p2) {};

    % Draw lune for (p4, p5)
    \path let \p1=(p4), \p2=(p5) in
        node[draw, circle, blue, dashed, minimum size={2*veclen(\x2-\x1,\y2-\y1)}] at (\p1) {};
    \path let \p1=(p4), \p2=(p5) in
        node[draw, circle, blue, dashed, minimum size={2*veclen(\x2-\x1,\y2-\y1)}] at (\p2) {};

\end{tikzpicture}
